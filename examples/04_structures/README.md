# Structures

Examples that define small mathematical structures and structure-like interfaces in Litex.

Each Litex block below is checked independently by `cargo test run_examples -- --nocapture`.
The `Category` and `System surface` fields say what part of Litex the example is meant to exercise.

## 1. `abstract_integral`

- Category: `case study`
- System surface: `abstract_prop plus template`
- Purpose: Shows an abstract integrability interface.

```litex
abstract_prop has_integral(s, f, value)

prop integrable(s set, f set):
    $restricts_to(f, s)
    exist a R st {$has_integral(s, f, a)}

template<s power_set(R), f set: $integrable(s, f)>:
    have by exist a R st {$has_integral(s, f, a)}: integral

know forall s power_set(R), f, g set:
    $integrable(s, f)
    $integrable(s, g)
    =>:
        $integrable(s, '(x s)R{f(x) + g(x)})

        \integral<s, f> + \integral<s, g> = \integral<s, '(x s)R{f(x) + g(x)}>
```

## 2. `diagonal_matrix`

- Category: `obj`
- System surface: `matrix-like functions`
- Purpose: Shows a diagonal-matrix structure interface.

```litex
prop is_diagonal_matrix(n N_pos, m fn(i closed_range(1, n), j closed_range(1, n)) R):
    forall i closed_range(1, n), j closed_range(1, n):
        i != j
        =>:
            m(i, j) = 0

claim:
    prove:
        forall M fn(i closed_range(1, 3), j closed_range(1, 3)) R:
            M(1, 1) = 1
            M(1, 2) = 0
            M(1, 3) = 0
            M(2, 1) = 0
            M(2, 2) = 1
            M(2, 3) = 0
            M(3, 1) = 0
            M(3, 2) = 0
            M(3, 3) = 1
            =>:
                $is_diagonal_matrix(3, M)
    
    by for:
        prove:
            forall i closed_range(1, 3), j closed_range(1, 3):
                i != j
                =>:
                    M(i, j) = 0
```

## 3. `group`

- Category: `case study`
- System surface: `group structure`
- Purpose: Shows a small group-style structure.

```litex
prop group_property(s set, zero s, add fn(x, y s) s, inv fn(x s) s):
    forall x, y, z s:
        add(x, add(y, z)) = add(add(x, y), z)
    forall x s:
        add(x, zero) = x
        add(zero, x) = x
        add(x, inv(x)) = zero
        add(inv(x), x) = zero

struct group<s set>:
    zero s
    add fn(x, y s) s
    inv fn(x s) s
    <=>:
        $group_property(s, zero, add, inv)
```

## 4. `group_property`

- Category: `case study`
- System surface: `group property interfaces`
- Purpose: Shows abstract group-property APIs and proof debt boundaries.

```litex
# Reusable group-theory property interface.

prop group_property(G set, one G, mul fn(x, y G) G, inv fn(x G) G):
    forall x, y, z G:
        mul(x, mul(y, z)) = mul(mul(x, y), z)
    forall x G:
        mul(one, x) = x
        mul(x, one) = x
        mul(x, inv(x)) = one
        mul(inv(x), x) = one

prop injective_fn(S, T set, phi fn(x S) T):
    forall x1, x2 S:
        phi(x1) = phi(x2)
        =>:
            x1 = x2

prop surjective_fn(S, T set, phi fn(x S) T):
    forall y T:
        exist x S st {y = phi(x)}

prop bijective_fn(S, T set, phi fn(x S) T):
    $injective_fn(S, T, phi)
    $surjective_fn(S, T, phi)

abstract_prop group_homomorphism(G, H, phi)
abstract_prop generator_map(G, H, a, b, c, d, phi)
abstract_prop generated_by_two_elements(G, a, b)

prop group_isomorphism(G, H set, phi fn(x G) H):
    $group_homomorphism(G, H, phi)
    $bijective_fn(G, H, phi)

prop isomorphic_group(G, H set):
    exist phi fn(x G) H st {$group_isomorphism(G, H, phi)}

abstract_prop two_generator_presentation(G, a, b)
abstract_prop target_satisfies_presentation_relators(G, H, a, b, c, d)

prop presentation_universal_map(G, H set, a, b G, c, d H):
    $two_generator_presentation(G, a, b)
    $target_satisfies_presentation_relators(G, H, a, b, c, d)
    exist phi fn(x G) H st {$group_homomorphism(G, H, phi), phi(a) = c, phi(b) = d, $generator_map(G, H, a, b, c, d, phi)}

# A target pair satisfying the relators receives the canonical homomorphism.
know forall G, H set, a, b G, c, d H:
    $two_generator_presentation(G, a, b)
    $target_satisfies_presentation_relators(G, H, a, b, c, d)
    =>:
        $presentation_universal_map(G, H, a, b, c, d)

abstract_prop image_contains_two_generator_images(G, H, a, b, c, d, phi)
abstract_prop image_contains_all_two_generator_words(G, H, c, d, phi)
abstract_prop image_equals_target_group(G, H, phi)

prop two_generator_homomorphic_image_surjective(G, H set, a, b G, c, d H, phi fn(x G) H):
    $group_homomorphism(G, H, phi)
    phi(a) = c
    phi(b) = d
    $generated_by_two_elements(H, c, d)
    $image_contains_two_generator_images(G, H, a, b, c, d, phi)
    $image_contains_all_two_generator_words(G, H, c, d, phi)
    $image_equals_target_group(G, H, phi)
    $surjective_fn(G, H, phi)

# If the target is generated by the images of the source generators, the map is onto.
know forall G, H set, a, b G, c, d H, phi fn(x G) H:
    $group_homomorphism(G, H, phi)
    phi(a) = c
    phi(b) = d
    $generated_by_two_elements(H, c, d)
    =>:
        $two_generator_homomorphic_image_surjective(G, H, a, b, c, d, phi)

abstract_prop finite_cardinality_at_most(S, n)
abstract_prop finite_cardinality_exactly(S, n)

prop has_at_most_16_elements(S set):
    $finite_cardinality_at_most(S, 16)

prop has_exactly_16_elements(S set):
    $finite_cardinality_exactly(S, 16)

abstract_prop finite_cardinality_comparison_for_map(S, T, phi, n)

prop surjective_same_finite_size_map_is_injective(S, T set, phi fn(x S) T, n N):
    $surjective_fn(S, T, phi)
    $finite_cardinality_at_most(S, n)
    $finite_cardinality_exactly(T, n)
    $finite_cardinality_comparison_for_map(S, T, phi, n)
    $injective_fn(S, T, phi)

prop finite_surjective_map_bijective_by_cardinality(S, T set, phi fn(x S) T, n N):
    $surjective_fn(S, T, phi)
    $finite_cardinality_at_most(S, n)
    $finite_cardinality_exactly(T, n)
    $surjective_same_finite_size_map_is_injective(S, T, phi, n)
    $bijective_fn(S, T, phi)

# A surjection from a set of size at most n onto a set of size n is bijective.
know forall S, T set, phi fn(x S) T, n N:
    $surjective_fn(S, T, phi)
    $finite_cardinality_at_most(S, n)
    $finite_cardinality_exactly(T, n)
    =>:
        $finite_surjective_map_bijective_by_cardinality(S, T, phi, n)

abstract_prop every_element_has_normal_form(G, slots)
abstract_prop normal_form_uses_slot_set(G, slots)
abstract_prop slot_set_has_n_slots(slots, n)
abstract_prop normal_form_slot_map_bounds_cardinality(G, slots, n)

prop normal_form_cardinality_bound(G, slots set, n N):
    $every_element_has_normal_form(G, slots)
    $normal_form_uses_slot_set(G, slots)
    $slot_set_has_n_slots(slots, n)
    $normal_form_slot_map_bounds_cardinality(G, slots, n)
    $finite_cardinality_at_most(G, n)

# A normal form landing in n slots bounds the group's cardinality by n.
know forall G, slots set, n N:
    $every_element_has_normal_form(G, slots)
    $normal_form_uses_slot_set(G, slots)
    $slot_set_has_n_slots(slots, n)
    =>:
        $normal_form_cardinality_bound(G, slots, n)

abstract_prop dihedral_normal_forms_cover(G, r, f)
abstract_prop dihedral_normal_forms_use_16_slots(G, r, f)

prop dihedral_normal_form_cardinality_bound(G set, r, f G):
    $dihedral_normal_forms_cover(G, r, f)
    $dihedral_normal_forms_use_16_slots(G, r, f)
    $has_at_most_16_elements(G)

# Dihedral normal forms r^i or r^i f with i in {0,...,7} give at most 16 elements.
know forall G set, r, f G:
    $dihedral_normal_forms_cover(G, r, f)
    $dihedral_normal_forms_use_16_slots(G, r, f)
    =>:
        $dihedral_normal_form_cardinality_bound(G, r, f)
```

## 5. `group_quotient`

- Category: `case study`
- System surface: `quotient group construction`
- Purpose: Shows a larger quotient group interface and proof development.

```litex
# Quotient set and quotient multiplication for a normal subgroup.

prop GroupProperty(s nonempty_set, inv fn(x s) s, op fn(x, y s) s, e s):
    forall x, y, z s:
        op(x, op(y, z)) = op(op(x, y), z)
    forall x s:
        op(e, x) = x
        op(x, e) = x
    forall x s:
        op(x, inv(x)) = e
        op(inv(x), x) = e

struct Group<s nonempty_set>:
    inv fn(x s) s
    op fn(x, y s) s
    e s
    <=>:
        $GroupProperty(s, inv, op, e)

macro G "&Group<s>{g}"

prop is_subgroup(s nonempty_set, g &Group<s>, h power_set(s)):
    @G.e $in h
    forall a, b s:
        a $in h
        b $in h
        =>:
            @G.op(a, b) $in h
    forall a s:
        a $in h
        =>:
            @G.inv(a) $in h

prop is_normal_subgroup(s nonempty_set, g &Group<s>, h power_set(s)):
    $is_subgroup(s, g, h)
    forall x, a s:
        a $in h
        =>:
            @G.op(@G.op(x, a), @G.inv(x)) $in h

prop is_left_coset_with_representative(s nonempty_set, g &Group<s>, h power_set(s), x s, c power_set(s)):
    forall y s:
        =>:
            y $in c
        <=>:
            exist a s st {a $in h, y = @G.op(x, a)}

prop is_in_left_coset_by_representative(s nonempty_set, g &Group<s>, h power_set(s), x, y s):
    exist a s st {a $in h, y = @G.op(x, a)}

prop is_left_coset(s nonempty_set, g &Group<s>, h power_set(s), c power_set(s)):
    exist x s st {$is_left_coset_with_representative(s, g, h, x, c)}

prop is_group_quotient_set(s nonempty_set, g &Group<s>, h power_set(s), q power_set(power_set(s))):
    q = {c power_set(s): $is_left_coset(s, g, h, c)}

claim:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s):
            exist! q power_set(power_set(s)) st {$is_group_quotient_set(s, g, h, q)}
    witness exist! q power_set(power_set(s)) st {$is_group_quotient_set(s, g, h, q)} from {c power_set(s): $is_left_coset(s, g, h, c)}:
        $is_group_quotient_set(s, g, h, {c power_set(s): $is_left_coset(s, g, h, c)})

template<s nonempty_set>:
    have fn group_quotient as set:
        prove:
            forall g &Group<s>, h power_set(s):
                exist! q power_set(power_set(s)) st {$is_group_quotient_set(s, g, h, q)}

prop is_quotient_product_coset(s nonempty_set, g &Group<s>, h power_set(s), c1 power_set(s), c2 power_set(s), c3 power_set(s)):
    forall x, y s:
        $is_left_coset_with_representative(s, g, h, x, c1)
        $is_left_coset_with_representative(s, g, h, y, c2)
        =>:
            $is_left_coset_with_representative(s, g, h, @G.op(x, y), c3)

prop quotient_product_representatives_equal(s nonempty_set, g &Group<s>, h power_set(s), q power_set(power_set(s)), c1, c2 q, x1, x2, y1, y2 s, out1, out2 q):
    out1 = out2

prop quotient_product_well_defined(s nonempty_set, g &Group<s>, h power_set(s), q power_set(power_set(s))):
    forall c1, c2 q, x1, x2, y1, y2 s, out1, out2 q:
        $is_left_coset_with_representative(s, g, h, x1, c1)
        $is_left_coset_with_representative(s, g, h, x2, c1)
        $is_left_coset_with_representative(s, g, h, y1, c2)
        $is_left_coset_with_representative(s, g, h, y2, c2)
        $is_left_coset_with_representative(s, g, h, @G.op(x1, y1), out1)
        $is_left_coset_with_representative(s, g, h, @G.op(x2, y2), out2)
        =>:
            $quotient_product_representatives_equal(s, g, h, q, c1, c2, x1, x2, y1, y2, out1, out2)

prop is_quotient_multiplication(s nonempty_set, g &Group<s>, h power_set(s), q power_set(power_set(s)), quotient_mul fn(c1, c2 q) q):
    forall c1, c2 q:
        $is_quotient_product_coset(s, g, h, c1, c2, quotient_mul(c1, c2))

thm group_left_cancel:
    prove:
        forall s nonempty_set, g &Group<s>, a, b, c s:
            @G.op(a, b) = @G.op(a, c)
            =>:
                b = c
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    @G.op(@G.inv(a), a) = @G.e
    @G.op(@G.e, b) = b
    @G.op(@G.e, c) = c
    @G.op(@G.op(@G.inv(a), a), b) = @G.op(@G.e, b) = b
    @G.op(@G.op(@G.inv(a), a), b) = @G.op(@G.inv(a), @G.op(a, b)) = @G.op(@G.inv(a), @G.op(a, c)) = @G.op(@G.op(@G.inv(a), a), c) = @G.op(@G.e, c) = c
    b = c

thm group_inv_inv:
    prove:
        forall s nonempty_set, g &Group<s>, a s:
            @G.inv(@G.inv(a)) = a
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    @G.op(@G.inv(a), @G.inv(@G.inv(a))) = @G.e
    @G.op(@G.inv(a), a) = @G.e
    @G.op(@G.inv(a), @G.inv(@G.inv(a))) = @G.op(@G.inv(a), a)
    by thm group_left_cancel(s, g, @G.inv(a), @G.inv(@G.inv(a)), a)

thm normal_subgroup_inv_conjugate:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), y, a s:
            $is_normal_subgroup(s, g, h)
            a $in h
            =>:
                @G.op(@G.op(@G.inv(y), a), y) $in h
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    by thm group_inv_inv(s, g, y)
    @G.op(@G.op(@G.inv(y), a), @G.inv(@G.inv(y))) $in h
    @G.op(@G.op(@G.inv(y), a), @G.inv(@G.inv(y))) = @G.op(@G.op(@G.inv(y), a), y)
    @G.op(@G.op(@G.inv(y), a), y) $in h

thm left_coset_contains_representative:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), x s, c power_set(s):
            $is_subgroup(s, g, h)
            $is_left_coset_with_representative(s, g, h, x, c)
            =>:
                x $in c
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    h $in power_set(s)
    @G.e $in h
    @G.e $in s
    @G.op(x, @G.e) = x
    x = @G.op(x, @G.e)
    witness exist a s st {a $in h, x = @G.op(x, a)} from @G.e:
        @G.e $in h
        x = @G.op(x, @G.e)
    x $in c

thm same_left_coset_has_factor:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), c power_set(s), x, y s:
            $is_subgroup(s, g, h)
            $is_left_coset_with_representative(s, g, h, x, c)
            $is_left_coset_with_representative(s, g, h, y, c)
            =>:
                exist a s st {a $in h, y = @G.op(x, a)}
    by thm left_coset_contains_representative(s, g, h, y, c)
    y $in c
    exist a s st {a $in h, y = @G.op(x, a)}

thm same_representative_left_cosets_equal:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), x s, c, d power_set(s):
            $is_left_coset_with_representative(s, g, h, x, c)
            $is_left_coset_with_representative(s, g, h, x, d)
            =>:
                c = d
    by extension:
        prove:
            c = d
        forall y c:
            y $in c
            exist a s st {a $in h, y = @G.op(x, a)}
            y $in d
        forall y d:
            y $in d
            exist a s st {a $in h, y = @G.op(x, a)}
            y $in c

thm left_coset_change_representative:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), x, y s, c power_set(s), a s:
            $is_subgroup(s, g, h)
            $is_left_coset_with_representative(s, g, h, x, c)
            a $in h
            y = @G.op(x, a)
            =>:
                $is_left_coset_with_representative(s, g, h, y, c)
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    h $in power_set(s)
    @G.inv(a) $in h
    claim:
        prove:
            forall z s:
                z $in c
                =>:
                    exist b s st {b $in h, z = @G.op(y, b)}
        have by exist b s st {b $in h, z = @G.op(x, b)}: b
        @G.op(@G.inv(a), b) $in h
        @G.op(a, @G.inv(a)) = @G.e
        @G.op(@G.e, b) = b
        @G.op(a, @G.op(@G.inv(a), b)) = @G.op(@G.op(a, @G.inv(a)), b)
        @G.op(x, @G.op(a, @G.op(@G.inv(a), b))) = @G.op(x, @G.op(@G.op(a, @G.inv(a)), b))
        @G.op(y, @G.op(@G.inv(a), b)) = @G.op(@G.op(x, a), @G.op(@G.inv(a), b)) = @G.op(x, @G.op(a, @G.op(@G.inv(a), b))) = @G.op(x, @G.op(@G.op(a, @G.inv(a)), b)) = @G.op(x, @G.op(@G.e, b)) = @G.op(x, b) = z
        z = @G.op(y, @G.op(@G.inv(a), b))
        witness exist t s st {t $in h, z = @G.op(y, t)} from @G.op(@G.inv(a), b):
            @G.op(@G.inv(a), b) $in h
            z = @G.op(y, @G.op(@G.inv(a), b))
    claim:
        prove:
            forall z s:
                exist b s st {b $in h, z = @G.op(y, b)}
                =>:
                    z $in c
        have by exist b s st {b $in h, z = @G.op(y, b)}: b
        @G.op(a, b) $in h
        @G.op(y, b) = @G.op(@G.op(x, a), b) = @G.op(x, @G.op(a, b))
        z = @G.op(x, @G.op(a, b))
        witness exist t s st {t $in h, z = @G.op(x, t)} from @G.op(a, b):
            @G.op(a, b) $in h
            z = @G.op(x, @G.op(a, b))
        z $in c
    $is_left_coset_with_representative(s, g, h, y, c)

thm group_move_across:
    prove:
        forall s nonempty_set, g &Group<s>, y, u, k s:
            k = @G.op(@G.op(@G.inv(y), u), y)
            =>:
                @G.op(u, y) = @G.op(y, k)
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    @G.op(y, @G.inv(y)) = @G.e
    @G.op(@G.e, u) = u
    @G.op(y, @G.op(@G.inv(y), u)) = @G.op(@G.op(y, @G.inv(y)), u)
    @G.op(@G.op(y, @G.op(@G.inv(y), u)), y) = @G.op(@G.op(@G.op(y, @G.inv(y)), u), y)
    @G.op(y, k) = @G.op(y, @G.op(@G.op(@G.inv(y), u), y)) = @G.op(@G.op(y, @G.op(@G.inv(y), u)), y) = @G.op(@G.op(@G.op(y, @G.inv(y)), u), y) = @G.op(@G.op(@G.e, u), y) = @G.op(u, y)
    @G.op(u, y) = @G.op(y, k)

thm quotient_product_has_factor:
    prove:
        forall s nonempty_set, g &Group<s>, h power_set(s), x1, x2, y1, y2, u, v s:
            $is_normal_subgroup(s, g, h)
            u $in h
            v $in h
            x2 = @G.op(x1, u)
            y2 = @G.op(y1, v)
            =>:
                exist a s st {a $in h, @G.op(x2, y2) = @G.op(@G.op(x1, y1), a)}
    $GroupProperty(s, @G.inv, @G.op, @G.e)
    have k set = @G.op(@G.op(@G.inv(y1), u), y1)
    k $in s
    by thm normal_subgroup_inv_conjugate(s, g, h, y1, u)
    k $in h
    @G.op(k, v) $in h
    by thm group_move_across(s, g, y1, u, k)
    @G.op(u, y1) = @G.op(y1, k)
    @G.op(u, @G.op(y1, v)) = @G.op(@G.op(u, y1), v)
    @G.op(x1, @G.op(u, @G.op(y1, v))) = @G.op(x1, @G.op(@G.op(u, y1), v))
    @G.op(x1, @G.op(@G.op(u, y1), v)) = @G.op(x1, @G.op(@G.op(y1, k), v))
```
